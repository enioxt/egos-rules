json_get() {

key = sys.argv[1].lstrip('.').split('.')
try:
    d = json.load(sys.stdin)
    for k in key: d = d[k]
    print(d if d is not None else 'null')
except Exception:
    print('null')
" "$_jg_key" 2>/dev/null
  else
    echo "null"
  fi
