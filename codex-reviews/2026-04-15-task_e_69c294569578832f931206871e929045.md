+

+import type { HealthCheck, HealthCheckResult, HealthStatus } from "./HealthCheck.js";
+
+export interface AggregatedHealth {
+  status: HealthStatus;
+  timestamp: string;
+  checks: HealthCheckResult[];
+}
+
+export async function runHealthChecks(checks: HealthCheck[]): Promise<AggregatedHealth> {
+  const results = await Promise.all(checks.map((check) => check.check()));
+  const hasCriticalFailure = results.some((result) => result.critical && result.status === "unhealthy");
+  const hasDegraded = results.some((result) => result.status === "degraded");
+
+  return {
+    status: hasCriticalFailure ? "unhealthy" : hasDegraded ? "degraded" : "healthy",
+    timestamp: new Date().toISOString(),
+    checks: results,
+  };
+}
