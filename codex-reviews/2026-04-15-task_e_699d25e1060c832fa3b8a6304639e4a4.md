+import './ToolsCatalog.css'
+import { TOOLS_CATALOG, TOOL_STATUS_LABEL, type ToolStatus } from './toolsCatalogData'
+
+type SortMode = 'status' | 'maturity-desc' | 'name-asc'
+
+const STATUS_RANK: Record<ToolStatus, number> = { live: 0, beta: 1, planned: 2 }
+
+export default function ToolsCatalog() {
+  const [query, setQuery] = useState('')
+  const [status, setStatus] = useState<'all' | ToolStatus>('all')
+  const [sort, setSort] = useState<SortMode>('status')
+
+  const categories = useMemo(() => ['all', ...new Set(TOOLS_CATALOG.map((tool) => tool.category))], [])
+  const [category, setCategory] = useState<string>('all')
+
+  const summary = useMemo(() => {
+    const live = TOOLS_CATALOG.filter((tool) => tool.status === 'live').length
+    const beta = TOOLS_CATALOG.filter((tool) => tool.status === 'beta').length
+    const planned = TOOLS_CATALOG.filter((tool) => tool.status === 'planned').length
+    const mcpLive = TOOLS_CATALOG.filter((tool) => tool.category === 'MCP' && tool.status === 'live').length
+    return { total: TOOLS_CATALOG.length, live, beta, planned, mcpLive }
+  }, [])
+
+  const tools = useMemo(() => {
+    const q = query.trim().toLowerCase()
+    const filtered = TOOLS_CATALOG.filter((tool) => {
+      const byStatus = status === 'all' || tool.status === status
+      const byCategory = category === 'all' || tool.category === category
+      const byQuery = !q || [tool.name, tool.short, tool.description, ...tool.tags].join(' ').toLowerCase().includes(q)
+      return byStatus && byCategory && byQuery
+    })
+
+    const ordered = [...filtered]
+    if (sort === 'status') {
+      ordered.sort((a, b) => STATUS_RANK[a.status] - STATUS_RANK[b.status] || b.maturity - a.maturity)
+    } else if (sort === 'maturity-desc') {
+      ordered.sort((a, b) => b.maturity - a.maturity)
+    } else {
+      ordered.sort((a, b) => a.name.localeCompare(b.name, 'pt-BR'))
+    }
+
+    return ordered
+  }, [query, status, category, sort])
+
+  return (
+    <main className="tools-page">
+      <section className="tools-hero">
+        <h1>Todas as Ferramentas EGOS</h1>
+        <p>Catálogo único com status real, filtros e acesso rápido para cada ferramenta.</p>
+        <div className="tools-kpis">
+          <span><strong>{summary.total}</strong> total</span>
+          <span><strong>{summary.live}</strong> funcionando</span>
+          <span><strong>{summary.beta}</strong> beta</span>
+          <span><strong>{summary.planned}</strong> planejadas</span>
+          <span><strong>{summary.mcpLive}</strong> MCP ativas</span>
+        </div>
+      </section>
+
+      <section className="tools-filters">
+        <label className="tools-search">
+          <Search size={14} />
+          <input
+            value={query}
+            onChange={(event) => setQuery(event.target.value)}
+            placeholder="Buscar por nome, tag ou descrição"
+          />
+        </label>
+
+        <label>
+          <Filter size={14} />
+          <select value={status} onChange={(event) => setStatus(event.target.value as 'all' | ToolStatus)}>
+            <option value="all">Todos status</option>
+            <option value="live">Funcionando</option>
+            <option value="beta">Beta</option>
+            <option value="planned">Planejada</option>
+          </select>
+        </label>
+
+        <label>
+          Categoria
+          <select value={category} onChange={(event) => setCategory(event.target.value)}>
+            {categories.map((value) => (
+              <option key={value} value={value}>{value === 'all' ? 'Todas categorias' : value}</option>

+          </select>
+        </label>
+
+        <label>

+            <option value="maturity-desc">Maturidade (desc)</option>

+          </select>
+        </label>
+      </section>

+      <section className="tools-grid">
+        {tools.map((tool) => (
+          <article key={tool.id} className={`tool-card status-${tool.status}`}>
+            <header>
+              <span className="tool-status">{TOOL_STATUS_LABEL[tool.status]}</span>
+              <span className="tool-maturity">{tool.maturity}%</span>
+            </header>
+            <h2>{tool.name}</h2>
+            <p>{tool.short}</p>
+            <div className="tool-tags">{tool.tags.map((tag) => <span key={tag}>{tag}</span>)}</div>
+            <footer>
+              <Link to={`/tools/${tool.slug}`} className="tool-btn">Ver detalhes</Link>

+  name: "booking_start" | "booking_reserved" | "booking_confirmed" | "booking_no_show" | "booking_reactivated";
+  leadId: string;
+  bookingId?: string;
+  at: string;

+  ok: boolean;
+  tool: string;
+  data?: T;
+  error?: string;

+  serviceId: string;

+  phoneE164: string;

+  displayName?: string;
+  timezone?: string;

+type CoreState = {
+  leads: Map<string, Lead>;
+  slots: Map<string, Slot>;
+  bookings: Map<string, Booking>;
+  events: MCPEvent[];
+};
+
+export function createCoreState(): CoreState {
+  const slots = new Map<string, Slot>();
+  const now = new Date();
+  for (let i = 0; i < 6; i++) {
+    const start = new Date(now.getTime() + (i + 1) * 60 * 60 * 1000);
+    const end = new Date(start.getTime() + 45 * 60 * 1000);
+    const slotId = `slot_${i + 1}`;
+    slots.set(slotId, {
+      slotId,
+      startAt: start.toISOString(),
+      endAt: end.toISOString(),
+      timezone: "America/Sao_Paulo",
+      available: true,
+    });
+  }
+
+  return {
+    leads: new Map<string, Lead>(),
+    slots,
+    bookings: new Map<string, Booking>(),
+    events: [],
+  };
+}

+const state = createCoreState();

+function ok<T>(tool: string, data: T): ToolResponse<T> {
+  return { ok: true, tool, data };
+}
+
+function fail(tool: string, error: string): ToolResponse {
+  return { ok: false, tool, error };
+}
+
+function hasUnexpectedKeys(input: Record<string, unknown>, allowed: string[]): boolean {
+  return Object.keys(input).some((key) => !allowed.includes(key));
+}
+
+
+function emitEvent(core: CoreState, event: MCPEvent) {
+  core.events.push(event);
+}

+export function getEvents(core: CoreState = state): MCPEvent[] {
+  return [...core.events];

+function findSlots(input: FindSlotsInput, core: CoreState = state): ToolResponse<{ slots: Slot[] }> {
+  if (hasUnexpectedKeys(input as Record<string, unknown>, ["serviceId", "windowStart", "windowEnd", "timezone", "limit"])) {
+    return fail("booking.find-slots", "invalid input");
+  }
+  if (!input.timezone || !input.serviceId || !input.windowStart || !input.windowEnd) {
+    return fail("booking.find-slots", "invalid input");
+  }
+
+  const start = new Date(input.windowStart).getTime();
+  const end = new Date(input.windowEnd).getTime();
+  if (!Number.isFinite(start) || !Number.isFinite(end) || start >= end) {
+    return fail("booking.find-slots", "invalid window");
+  }
+
+  const slots = [...core.slots.values()]
+    .filter((slot) => {
+      if (!slot.available || slot.timezone !== input.timezone) return false;
+      const slotStart = new Date(slot.startAt).getTime();
+      return slotStart >= start && slotStart <= end;
+    })
+    .slice(0, input.limit ?? 20);
+  emitEvent(core, { name: "booking_start", leadId: "anonymous", at: new Date().toISOString() });
+  return ok("booking.find-slots", { slots });
+}

+function reserveSlot(input: ReserveSlotInput, core: CoreState = state): ToolResponse<{ booking: Booking }> {
+  if (hasUnexpectedKeys(input as Record<string, unknown>, ["leadId", "serviceId", "slotStart", "slotEnd", "timezone", "notes"])) {
+    return fail("booking.reserve-slot", "invalid input");
+  }
+  if (!input.leadId || !input.serviceId || !input.slotStart || !input.slotEnd || !input.timezone) {
+    return fail("booking.reserve-slot", "invalid input");
+  }
+  const slot = [...core.slots.values()].find((row) =>
+    row.available &&
+    row.timezone === input.timezone &&
+    row.startAt === input.slotStart &&
+    row.endAt === input.slotEnd
+  );
+  if (!slot) {
+    return fail("booking.reserve-slot", "slot unavailable");
+  }
+
+  const bookingId = `bk_${slot.slotId}_${input.leadId}`;
+  slot.available = false;
+  core.slots.set(slot.slotId, slot);
+
+  const booking: Booking = {
+    bookingId,
+    leadId: input.leadId,
+    slotId: slot.slotId,
+    serviceId: input.serviceId,
+    status: "reserved",
+  };
+  core.bookings.set(bookingId, booking);
+  emitEvent(core, { name: "booking_reserved", leadId: input.leadId, bookingId, at: new Date().toISOString() });
+
+  return ok("booking.reserve-slot", { booking });

+#!/usr/bin/env bun
+import { existsSync, readFileSync } from 'node:fs'
+import { resolve } from 'node:path'
+
+type EventDef = {
+  name: string
+  stage: number
+  requiredProperties: string[]
+}
+
+type FunnelSpec = {
+  version: string
+  domain: string
+  funnel: string
+  requiredCommonProperties: string[]
+  events: EventDef[]
+}
+
+const SPEC_PATH = 'scripts/booking-funnel-events.json'
+const REQUIRED_EVENT_NAMES = [
+  'booking_start',
+  'booking_confirmed',
+  'booking_no_show',
+  'booking_reactivated',
+]
+
+function loadSpec(): FunnelSpec {
+  const abs = resolve(SPEC_PATH)
+  if (!existsSync(abs)) throw new Error(`Missing funnel spec file: ${SPEC_PATH}`)
+  return JSON.parse(readFileSync(abs, 'utf8')) as FunnelSpec
+}
+
+function hasDuplicates(values: string[]): boolean {
+  return new Set(values).size !== values.length
+}
+
+function run() {
+  const spec = loadSpec()
+  const issues: string[] = []
+
+  if (!spec.events.length) issues.push('events list must not be empty')
+  if (!spec.requiredCommonProperties.length) issues.push('requiredCommonProperties must not be empty')
+
+  const eventNames = spec.events.map((event) => event.name)
+  if (hasDuplicates(eventNames)) issues.push('duplicate event names detected')
+
+  for (const requiredName of REQUIRED_EVENT_NAMES) {
+    if (!eventNames.includes(requiredName)) {
+      issues.push(`missing required event: ${requiredName}`)
+    }
+  }
+
+  const stages = spec.events.map((event) => event.stage)
+  const sorted = [...stages].sort((a, b) => a - b)
+  for (let index = 0; index < sorted.length; index += 1) {
+    if (sorted[index] !== index + 1) {
+      issues.push(`stage sequence must be contiguous starting at 1 (expected ${index + 1}, got ${sorted[index] ?? 'none'})`)
+      break
+    }
+  }
+
+  for (const event of spec.events) {
+    if (!event.requiredProperties.length) {
+      issues.push(`${event.name}: requiredProperties must not be empty`)
+      continue
+    }
+
+    if (hasDuplicates(event.requiredProperties)) {
+      issues.push(`${event.name}: duplicate requiredProperties detected`)
+    }
+
+    for (const property of spec.requiredCommonProperties) {
+      if (event.requiredProperties.includes(property)) {
+        issues.push(`${event.name}: common property must not be duplicated in requiredProperties (${property})`)
+      }
+    }
+  }
+
+  console.log('🔎 Booking Funnel Contract Check\n')
+  console.log(`Spec: ${SPEC_PATH}`)
+  console.log(`Events: ${spec.events.length}`)
+  console.log(`Common properties: ${spec.requiredCommonProperties.length}\n`)
+
+  if (issues.length) {
+    console.log('⛔ Booking funnel contract issues detected:')
+    for (const issue of issues) console.log(` - ${issue}`)
+    process.exit(1)
+  }
+
+  console.log('✅ Booking funnel contract is valid. Exit code 0.')
+}
+
+run()

+#!/usr/bin/env bun
+import { existsSync, readdirSync, statSync } from "node:fs";
+import { join, relative } from "node:path";
+
+type DocEntry = { path: string; status: "canonical" | "reference" | "archive" | "superseded" };
+type Domain = { id: string; owner: string; command: string; metric: string; docs: DocEntry[] };
+type IndexFile = { managedFolders: string[]; allowedUnindexed?: string[]; domains: Domain[] };
+
+function walkMarkdown(dir: string): string[] {
+  const out: string[] = [];
+  for (const item of readdirSync(dir)) {
+    const full = join(dir, item);
+    const st = statSync(full);
+    if (st.isDirectory()) {
+      out.push(...walkMarkdown(full));
+      continue;
+    }
+    if (st.isFile() && full.endsWith(".md")) out.push(full.replaceAll("\\", "/"));
+  }
+  return out;
+}
+
+function isAllowed(path: string, prefixes: string[]): boolean {
+  return prefixes.some((prefix) => path.startsWith(prefix));
+}
+
+const indexPath = "scripts/docs-canonical-index.json";
+if (!existsSync(indexPath)) {
+  console.error(`❌ missing ${indexPath}`);
+  process.exit(1);
+}
+
+const index = (await Bun.file(indexPath).json()) as IndexFile;
+const errors: string[] = [];
+
+if (!existsSync("docs/README.md")) {
+  errors.push("missing docs/README.md canonical index");
+}
+
+const seenPaths = new Set<string>();
+for (const domain of index.domains) {
+  if (!domain.owner || !domain.command || !domain.metric) {
+    errors.push(`domain ${domain.id} must define owner/command/metric`);
+  }
+
+  const canonicals = domain.docs.filter((doc) => doc.status === "canonical");
+  const onlyArchived = domain.docs.every((doc) => doc.status === "archive" || doc.status === "superseded");
+  if (!onlyArchived && canonicals.length !== 1) {
+    errors.push(`domain ${domain.id} must have exactly 1 canonical doc (found ${canonicals.length})`);
+  }
+
+  for (const doc of domain.docs) {
+    if (!existsSync(doc.path)) {
+      errors.push(`domain ${domain.id} references missing doc: ${doc.path}`);
+    }
+
+    if (seenPaths.has(doc.path)) {
+      errors.push(`doc referenced in multiple domains: ${doc.path}`);
+    }
+    seenPaths.add(doc.path);
+  }
+}
+
+const managedDocs = index.managedFolders.flatMap((folder) => (existsSync(folder) ? walkMarkdown(folder) : []));
+for (const file of managedDocs) {
+  const rel = relative(process.cwd(), file).replaceAll("\\", "/");
+  if (!seenPaths.has(rel) && !isAllowed(rel, index.allowedUnindexed ?? [])) {
+    errors.push(`unindexed managed doc: ${rel}`);
+  }
+}
+
+if (errors.length > 0) {
+  console.error("❌ Docs SSOT lint failed:\n");
+  for (const err of errors) console.error(` - ${err}`);
+  process.exit(1);
+}
+
+console.log(`✅ Docs SSOT lint passed (${index.domains.length} domains, ${seenPaths.size} indexed docs)`);

+type JsonSchema = {
+  type?: string | string[]
+  required?: string[]
+  properties?: Record<string, JsonSchema>
+  additionalProperties?: boolean | JsonSchema
+  items?: JsonSchema | JsonSchema[]
+}

+type DriftTarget = {
+  name: string
+  canonical: string
+  projection: string
+}
+
+type SsotMap = {
+  schemaTargets: DriftTarget[]
+}
+
+const MAP_PATH = 'scripts/ssot-map.json'
+
+function loadSsotMap(): SsotMap {
+  const abs = resolve(MAP_PATH)
+  if (!existsSync(abs)) throw new Error(`Missing SSOT map file: ${MAP_PATH}`)
+  const parsed = JSON.parse(readFileSync(abs, 'utf8')) as SsotMap
+  if (!parsed.schemaTargets?.length) throw new Error(`SSOT map has no schemaTargets: ${MAP_PATH}`)
+  return parsed
+}
+
+function loadSchema(path: string): JsonSchema {
+  const abs = resolve(path)
+  if (!existsSync(abs)) throw new Error(`Missing schema file: ${path}`)
+  return JSON.parse(readFileSync(abs, 'utf8')) as JsonSchema
+}
+
+function normalizeType(t: JsonSchema['type']): string {
+  if (Array.isArray(t)) return [...t].sort().join('|')
+  return t ?? 'undefined'
+}
+
+function diffSchema(canonical: JsonSchema, projection: JsonSchema, path = '$'): string[] {
+  const issues: string[] = []
+
+  if (normalizeType(canonical.type) !== normalizeType(projection.type)) {
+    issues.push(`${path}: type mismatch canonical=${normalizeType(canonical.type)} projection=${normalizeType(projection.type)}`)
+  }
+
+  const cReq = [...(canonical.required ?? [])].sort()
+  const pReq = [...(projection.required ?? [])].sort()
+  if (JSON.stringify(cReq) !== JSON.stringify(pReq)) {
+    issues.push(`${path}: required mismatch canonical=[${cReq.join(',')}] projection=[${pReq.join(',')}]`)
+  }
+
+  const cAdditional = canonical.additionalProperties
+  const pAdditional = projection.additionalProperties
+  const cAdditionalNorm = typeof cAdditional === 'object' ? 'object-schema' : String(cAdditional)
+  const pAdditionalNorm = typeof pAdditional === 'object' ? 'object-schema' : String(pAdditional)
+  if (cAdditionalNorm !== pAdditionalNorm) {
+    issues.push(`${path}: additionalProperties mismatch canonical=${cAdditionalNorm} projection=${pAdditionalNorm}`)
+  }
+
+  const cProps = canonical.properties ?? {}
+  const pProps = projection.properties ?? {}
+
+  for (const key of Object.keys(cProps)) {
+    if (!pProps[key]) {
+      issues.push(`${path}.${key}: missing in projection`)
+      continue
+    }
+    issues.push(...diffSchema(cProps[key], pProps[key], `${path}.${key}`))
+  }
+
+  for (const key of Object.keys(pProps)) {
+    if (!cProps[key]) {
+      issues.push(`${path}.${key}: extra in projection`)
+    }
+  }
+
+  if (canonical.items && projection.items) {
+    const cItems = Array.isArray(canonical.items) ? canonical.items[0] : canonical.items
+    const pItems = Array.isArray(projection.items) ? projection.items[0] : projection.items
+    if (cItems && pItems) {
+      issues.push(...diffSchema(cItems, pItems, `${path}[]`))
+    }
+  }
+
+  return issues
