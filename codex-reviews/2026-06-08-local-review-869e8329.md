// TG-FORMAT-002 (2026-06-03): o LLM e os comandos emitem markdown (*negrito*,
// _itálico_, `código`). Em texto plano isso virava lixo visual (asteriscos literais).
// Convertemos markdown → HTML do Telegram (renderiza de verdade); sendOne já cai
// para plain se o HTML falhar. Split ANTES de converter (não corta dentro de tag).
function mdToTelegramHtml(text: string): string {
  let s = text.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
  s = s.replace(/```([\s\S]*?)```/g, (_m, c) => `<pre>${String(c).trim()}</pre>`);
  s = s.replace(/`([^`\n]+)`/g, "<code>$1</code>");
  s = s.replace(/\*\*([^*\n]+)\*\*/g, "<b>$1</b>");
  s = s.replace(/\*([^*\n]+)\*/g, "<b>$1</b>");
  s = s.replace(/(^|[\s(])_([^_\n]+)_(?=[\s).,!?:;]|$)/g, "$1<i>$2</i>");
  return s;
}
