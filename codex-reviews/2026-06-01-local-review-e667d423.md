+def resolve_scihub_mirror(query_id):
+    """
+    Tries multiple Sci-Hub domains and URL formats to resolve a paper by DOI/PMID/URL.
+    Returns (pdf_url, domain)
+    """
+    clean_id = query_id.strip()
+    # Strip common DOI resolver prefixes
+    clean_id = re.sub(r'^https?://(dx\.)?doi\.org/', '', clean_id)
+    
+    url_formats = [
+        clean_id,
+        clean_id.replace('/', '%2F')  # Encoded version for mirrors that require it
+    ]
+
+    for domain in SCIHUB_DOMAINS:
+        for fmt in url_formats:
+            url = f"https://{domain}/{fmt}"
+            try:
+                if requests:
+                    response = requests.get(url, headers=HEADERS, timeout=10, allow_redirects=True)
+                    if response.status_code != 200:
+                        continue
+                    html = response.text
+                else:
+                    from urllib.request import Request, urlopen
+                    req = Request(url, headers=HEADERS)
+                    with urlopen(req, timeout=10) as resp:
+                        if resp.status != 200:
+                            continue
+                        html = resp.read().decode('utf-8', errors='ignore')
+
+                if not html or len(html) < 100:
+                    continue
+
+                # 1. Try citation_pdf_url meta tag
+                meta_match = re.search(r'<meta\s+name="citation_pdf_url"\s+content="([^"]+)"', html, re.IGNORECASE)
+                if meta_match:
+                    return meta_match.group(1), domain
+
+                # 2. Try object tag
+                object_match = re.search(r'<object[^>]+data="([^"]+)"', html, re.IGNORECASE)
+                if object_match:
+                    return object_match.group(1), domain
+
+                # 3. Try embed tag
+                embed_match = re.search(r'<embed[^>]+src="([^"]+)"', html, re.IGNORECASE)
+                if embed_match:
+                    return embed_match.group(1), domain
+
+                # 4. Try iframe tag
+                iframe_match = re.search(r'<iframe[^>]+src="([^"]+\.pdf[^"]*)"', html, re.IGNORECASE)
+                if iframe_match:
+                    return iframe_match.group(1), domain
+
+                # 5. Try general pdf href matching storage patterns
+                href_match = re.search(r'href="([^"]+/storage/[^"]+\.pdf[^"]*)"', html, re.IGNORECASE) or \

+            except Exception as e:
