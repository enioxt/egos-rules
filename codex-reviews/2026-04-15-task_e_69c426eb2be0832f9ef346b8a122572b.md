+### 18.1 Base legal prática para o cenário FORJA
+
+Para monitoramento interno de operação industrial (sem uso para marketing):
+- privilegiar base legal de **obrigação legal/regulatória**, **execução de contrato** e/ou **legítimo interesse** quando aplicável

+
+### 18.2 Regras obrigatórias de governança
+
+1. **Finalidade explícita e restrita** (segurança + eficiência operacional)
+2. **Minimização** (coletar só o necessário)
+3. **Transparência** (aviso interno + política de monitoramento)
+4. **Retenção definida** (prazo e descarte)
+5. **Controle de acesso** por perfil
+6. **Registro de tratamento** e resposta a direitos do titular
+7. **RIPD** para cenários de maior risco (ex.: biometria/reconhecimento facial)
+
+### 18.3 Sobre reconhecimento facial/biometria
+
+- Se houver identificação biométrica de pessoa, tratar como **alto risco** e com exigência de controles mais robustos

+find_egos_root() {

+    "${EGOS_ROOT:-}"
+    "$BASE_DIR/egos"
+    "/home/enio/egos"
+  )
+
+  for candidate in "${candidates[@]}"; do
+    if [[ -n "$candidate" && -d "$candidate" ]]; then
+      echo "$candidate"
+      return 0
+    fi
+  done
+
+  return 1

+repo_exists() {
+  local path="$1"
+  [[ -d "$path/.git" ]]
+}

+print_header() {
+  printf "\n============================================================\n"
+  printf "%s\n" "$1"
+  printf "============================================================\n"
+}
+
+run_start_probe() {
+  local repo_path="$1"
+  print_header "[START] $(basename "$repo_path")"
+
+  (
+    cd "$repo_path"
+
+    local branch
+    branch="$(git branch --show-current 2>/dev/null || echo 'N/A')"
+    local last
+    last="$(git log --oneline -1 2>/dev/null || echo 'N/A')"
+    local uncommitted
+    uncommitted="$(git status --short 2>/dev/null | wc -l | tr -d ' ')"
+
+    echo "Repo: $(basename "$repo_path") | Branch: $branch"
+    echo "Last commit: $last"
+    echo "Uncommitted: $uncommitted files"
+    echo "Recent commits:"
+    git log --oneline -5 2>/dev/null || true
+
+    local egos_root
+    if egos_root="$(find_egos_root)"; then
+      if [[ -x "$egos_root/scripts/sync-all-leaf-repos.sh" ]]; then
+        echo
+        echo "Sync check via $egos_root/scripts/sync-all-leaf-repos.sh --check"
+        bash "$egos_root/scripts/sync-all-leaf-repos.sh" --check 2>/dev/null | tail -10 || true
+      else
+        echo
+        echo "⚠️ Sync script not found in $egos_root/scripts"
+      fi
+    else
+      echo
+      echo "⚠️ EGOS root not found. Set EGOS_ROOT or clone egos as sibling repo."
+    fi
+
+    echo
+    echo "Required files check:"
+    local required=(
+      "AGENTS.md"
+      "TASKS.md"
+      "docs/CAPABILITY_REGISTRY.md"
+      ".windsurfrules"
+      ".guarani/PREFERENCES.md"
+      ".guarani/IDENTITY.md"
+      ".guarani/prompts/triggers.json"
+      "docs/SYSTEM_MAP.md"
+    )
+
+    for file in "${required[@]}"; do

+        echo "✅ $file"
+      else
+        echo "❌ $file"
+      fi
+    done
+
+    echo
+    echo "Meta-prompts check:"
+    [[ -f ".claude/commands/start.md" ]] && echo "✅ .claude/commands/start.md" || echo "❌ .claude/commands/start.md"
