+### 1. O Fluxo de Atendimento Nativo no WhatsApp (2026)

+Com a maturidade da API Oficial do WhatsApp e ferramentas como o **WhatsApp Flows**, a jornada pode ocorrer quase 100% no *backend*, mantendo o cliente no ambiente familiar do chat.

   146	function buildInspectionReceipt(
   147	  text: string,
   148	  output: string,
   149	  evidenceChain: EvidenceChain | undefined,
   150	  provenance: InspectProvenanceOptions | undefined,
   151	): InspectionReceipt {
   152	  const inspectedAt = new Date().toISOString();
   153	  const inputHash = sha256Text(text);
   154	  const outputHash = sha256Text(output);
   155	  let provenanceLevel: ProvenanceLevel = 'inspection_only';
   156	  let source: SourceReceipt | undefined;
   157	  if (provenance?.sourceUrl && provenance?.sourceMethod) {

   159	    const audit = provenance.rawRow
   160	      ? buildAuditFields({ rawRow: provenance.rawRow, sourceUrl: provenance.sourceUrl, method: provenance.sourceMethod, collectedAt: verifiedAt })
   161	      : undefined;
   162	    provenanceLevel = audit ? 'source_row_bound' : 'source_context';
   163	    source = {
   164	      ...audit,
   165	      source_url: provenance.sourceUrl.trim(),
   166	      source_method: provenance.sourceMethod.trim() || 'unknown',
   167	      verified_at: audit?.verified_at ?? verifiedAt,
   168	      source_fingerprint: audit?.source_fingerprint ?? sourceFingerprint(provenance.sourceUrl, provenance.sourceMethod, verifiedAt),
   169	      ...(provenance.query ? { queryHash: sha256Text(provenance.query) } : {}),
   170	      ...(provenance.recordId ? { recordId: provenance.recordId } : {}),
   171	      provenanceLevel,
   172	    };
   173	  }
